1240 class FixedFanConnectorBase : public UniformConnector {
1241 private:
1242  std::shared_ptr<RandomEngine> m_engine;
1243 
1244 protected:
1245  template <typename F>
1246  void generate_connections(size_t offs, size_t len, size_t subset_len,
1247  size_t i0, size_t i1, const F &f, bool self) const
1248  {
1249  // Initialize a vector with the identity permutation
1250  std::vector<size_t> perm(len);
1251  for (size_t i = 0; i < len; i++) {
1252  perm[i] = offs + i;
1253  }
1254 
1255  // Iterate over each target neuron, create a new permutation containing
1256  // the source neuron indices and instantiate the connection
1257  std::uniform_int_distribution<size_t> distr(0, perm.size() - 1);
1258  for (size_t i = i0; i < i1; i++) {
1259  // Generate a new permutation for the first subset_len elements
1260  for (size_t j = 0; j < subset_len; j++) {
1261  std::swap(perm[j], perm[distr(*m_engine)]);
1262  }
1263  // Create the connections
1264  if (self) {
1265  for (size_t j = 0; j < subset_len; j++) {
1266  f(i, perm[j]);
1267  }
1268  }
1269  else {
1270  size_t sub_len = subset_len;
1271  for (size_t j = 0; j < sub_len; j++) {
1272  if (i != perm[j]) {
1273  f(i, perm[j]);
1274  }
1275  else {
1276  sub_len = std::min(sub_len + 1, perm.size());
1277  }
1278  }
1279  }
1280  }
1281  }
1282

1285  std::shared_ptr<RandomEngine> engine,
